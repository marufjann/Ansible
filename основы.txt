ansible-inventory --list -i production.ini
ansible-inventory --list -i stage.ini
    # Эти команды полезны для того, чтобы убедиться, что инвентарь настроен правильно перед выполнением задач Ansible.
    # ansible-inventory --list:
        # Команда выводит инвентарь в формате JSON. 
        # По сути, это список всех хостов и групп хостов, с которыми Ansible может работать. 
        # В данном случае, формат будет JSON, который включает как групповые хосты, 
        # так и переменные, назначенные этим хостам.
    # -i production.ini
        # Опция -i указывает файл инвентаря. В этом случае это файл production.ini, 
        # который содержит инвентарь для среды продакшн. 
        # Он может включать в себя список серверов, 
        # IP-адресов и групп для работы Ansible в продакшн-среде.
ansible -i inventory/production.ini -m ping web
    # Команда, которую вы привели, используется для проверки доступности серверов в группе web из инвентарного файла production.ini
    # ansible:
        # Это команда для запуска Ansible. Она позволяет выполнять задачи на удаленных хостах.
    # -i inventory/production.ini:
        # Опция -i указывает путь к файлу инвентаря. В данном случае это файл production.ini, находящийся в директории inventory
    # -m ping:
        # Опция -m указывает модуль, который будет выполняться.В данном случае это модуль ping. Этот модуль используется для проверки доступности хостов
    # web:
        # Это группа хостов, как указано в инвентаре production.ini
ansible -i inventory/stage.ini -m setup web
    # -m setup:
        # Этот модуль собирает факты о системе хостов, такие как параметры операционной системы, процессора, памяти, сети и другие. 
        # Эти факты потом можно использовать в плейбуках для более тонкой настройки задач
ansible -i inventory/production.ini -m shell -a "hostname" web 
    # -m shell
        # Модуль shell в Ansible используется для выполнения произвольных команд в оболочке (shell) на удалённых хостах.
    # -a "hostname": аргумент, который передаёт команду для выполнения.
ansible -i inventory/hosts -m copy -a "src=/path/to/local/file dest=/path/to/remote/file" web
    # -m copy
        # Модуль copy копирует файл или директорию с хоста, где запускается Ansible, на один или несколько удалённых серверов. 
        # Можно указать права доступа, владельца и группу для копируемых файлов.
